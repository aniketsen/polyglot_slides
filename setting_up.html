Set up an SSH config file

  • To make it easier to connect to Marvin, we will set up an SSH config file on your laptop
  • This will allow us to connect to Marvin using a simple command like ssh marvin instead of ssh -i /path/to/your/private/key <username>@marvin.hpc.uni-bonn.de


Create a file named config in the ~/.ssh/ directory (if it doesn’t already exist)

And add the following content to the config file

Host marvin
  HostName marvin.hpc.uni-bonn.de
  User <username>
  IdentityFile /path/to/your/private/key (e.g., ~/.ssh/id_ed25519)
  IdentitiesOnly yes

torch_mm.py

import torch
import time
import argparse

N_warm = 100
N_bench = 1000

def benchmark_mm(N : int):
  A = torch.rand(N, N)
  B = torch.rand(N, N)
  # warmup
  for _ in range(N_warm):
    C = torch.mm(A, B)
  start_time = time.time()
  for _ in range(N_bench):
    C = torch.mm(A, B)
  end_time = time.time()
  avg_time = (end_time - start_time) / N_bench
  print(f"Average time for torch.mm with N={N}: {avg_time:.6f} seconds")

def main():
  # parse command line arguments
  parser = argparse.ArgumentParser(description="Benchmark torch.mm")
  parser.add_argument("--N", type=int, default=1024, help="Size of the matrices")
  args = parser.parse_args()
  benchmark_mm(args.N)

if __name__ == "__main__":
  main()

Define the benchmark_mm function which benchmarks the torch.mm function for multiplying two square matrices of size N x N

  • It creates two random matrices A and B of size N x N
  • It performs N_warm warmup iterations of matrix multiplication
  • It then performs N_bench benchmark iterations of matrix multiplication and times it with time.time()
  • Finally, it calculates and prints the average time taken for the matrix multiplication

torch_mm.py

import torch
import time
import argparse

N_warm = 100
N_bench = 1000

def benchmark_mm(N : int):
  A = torch.rand(N, N)
  B = torch.rand(N, N)
  # warmup
  for _ in range(N_warm):
    C = torch.mm(A, B)
  start_time = time.time()
  for _ in range(N_bench):
    C = torch.mm(A, B)
  end_time = time.time()
  avg_time = (end_time - start_time) / N_bench
  print(f"Average time for torch.mm with N={N}: {avg_time:.6f} seconds")

def main():
  # parse command line arguments
  parser = argparse.ArgumentParser(description="Benchmark torch.mm")
  parser.add_argument("--N", type=int, default=1024, help="Size of the matrices")
  args = parser.parse_args()
  benchmark_mm(args.N)

if __name__ == "__main__":
  main()

The main function parses the command line arguments to get the size of the matrices N and calls the benchmark_mm function with that size
